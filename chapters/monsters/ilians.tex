\subsection{Arnak}
\label{sec:monster:Arnak}

\begin{35e}{\Large{Arnak}}
  \srditem{Size/Type}{Humanoid (Arnak)}
  \srditem{Hit Dice}{8d8+8 (48 HP)}
  \srditem{Initiative}{+7}
  \srditem{Speed}{40 ft.}
  \srditem{Armour Class}{16 (+3 Dex, +3 Natural), touch 13, flat-footed 13}
  \srditem{BAB / Grapple}{+8/+16}
  \srditem{Attack}{Slam +8 1d4+4}
  \srditem{Full Attack}{2 Slams +8 1d4+4 and Bite +9 2d6+4 plus venom}
  \srditem{Space/Reach}{5 ft./5 ft.}
  \srditem{Special Attacks}{-}
  \srditem{Special Qualities}{Track, Scent (Ex), Blindsense (Ex), Hibernation
    (Ex), Arnak Spawn (Ex), Telepathic (Ex)
  }
  \srditem{Saves}{Fort +3, Ref +9, Will +4}
  \srditem{Abilities}{Str 18, Dex 16, Con 12, Int 8, Wis 13, Cha 8}
  \srditem{Skills}{Listen +7, Spot +6}
  \srditem{Feats}{Improved Initiative, Weapon Focus (Bite), Improved Grapple}
  \srditem{Environment}{Underground}
  \srditem{Organisation}{Solitary, pair or pack (3-5)}
  \srditem{Challenge Rating}{8}
  \srditem{Treasure}{No coins, normal goods, no items}
  \srditem{Alignment}{Any evil}
  \srditem{Venom (Ex)}{Arnak deliver venom with each bite. Fortitude save DC: 20
    negates, or otherwise the victim is paralysed for 1d4+1 rounds.
  }
  \srditem{Hibernation (Ex)}{Arnak loses Blindsense and Scent except when
    within 5 ft. when in hibernation beneath a stalactite. An Arnak in
    hibernation gains \emph{regeneration 3}.
  }
  \srditem{Arnak Spawn (Ex)}{If an arnak has a humanoid creature in a grapple
    or pin it can attempt to inject an \emph{Arnak Spawn} into the creature as
    a free action. A reflex slave DC 17 allows a grappled humanoid creature to
    shake off the spawn. If the victim is pinned or otherwise at the mercy of
    the Arnak no saving throw is allowed. An arnak spawn will drain its host
    for 1 point of constitution per day until 4d4 days have passed or the host
    dies prematurely of constitution drain. If the victim dies prematurely, so
    will the spawn. Once the spawn has matured, it will burst from the victims
    chest dealing 4d6 points of damage (Fortitude DC:16 half). Victims with an
    arnak spawn in their body are considered \emph{diseased}.
  }
\end{35e}
